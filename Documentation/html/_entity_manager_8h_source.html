45 // This function returns a vector full of all entityUIDs that possess the specified component
46 template<typename T>
47 std::vector<int> getEntitiesWithComponent(ComponentTypes componentType) {
48 std::vector<int> entitiesWithComponent;
49
50 for (const auto& [entityUID, entity] : entities) {
51 // Iterate over components of the entity
52 for (const auto& component : entity.components) {
53 if (component.first == componentType) {
54 entitiesWithComponent.push_back(entityUID);
55 break; // Found the component, no need to check further for this entity
56 }
57 }
58 }
59 return entitiesWithComponent;
60 }
61
62 // This function returns the actual component from the specified entityUID and component type
63 template<typename T>
64 T& getEntityComponent(int entityUID, const ComponentTypes componentType) {
65 auto& entity = getEntity(entityUID);
66 auto component = entity.components.find(componentType);
67 static_assert(std::is_base_of<Component, T>::value, "T must be a subclass of Component");
68 return dynamic_cast<T&>(*component->second);
69 }

73 void removeComponent(int entityUID, ComponentTypes componentType);
