13class FileTree {
14public:
15 FileTree() : selectedFilePath("") {}
16
17 void DisplayFileTree(const std::filesystem::path& path, CodeEditor& codeEditor) {
18 if (!std::filesystem::exists(path) || !std::filesystem::is_directory(path)) {
19 ImGui::Text("Invalid directory");
20 return;
21 }
22
23 ImGui::BeginChild("FileTree", ImVec2(0, 0), true);
24 for (const auto& entry : std::filesystem::directory_iterator(path)) {
25 const auto filename = entry.path().filename().string();
26 if (std::filesystem::is_directory(entry.status())) {
27 if (ImGui::TreeNode(filename.c_str())) {
28 DisplayFileTree(entry.path(), codeEditor); // Recursive call for subdirectories
29 ImGui::TreePop();
30 }
31 } else {
32 // Display selectable file names
33 if (ImGui::Selectable(filename.c_str())) {
34 selectedFilePath = entry.path().string();
35 }
36 // Handle double-click event
37 if (ImGui::IsItemHovered() && ImGui::IsMouseDoubleClicked(ImGuiMouseButton_Left)) {
38 codeEditor.LoadFile(selectedFilePath);
39 }
40 }
41 }
42 ImGui::EndChild();
43 }
44
45private:
46 std::string selectedFilePath;
47};
